      document.getElementById('upload-prompt').classList.remove('hidden');
      document.getElementById('file-info').classList.add('hidden');
      updateConfirmState();
    }

    function resetAll() {
      resetImage();
      document.getElementById('img-name').value = '';
    }

    function updateConfirmState() {
      document.getElementById('confirm-btn').disabled = !(currentFile && document.getElementById('img-name').value.trim());
    }

    function handleConfirm() {
      if (!currentFile) return;
      const name = document.getElementById('img-name').value.trim();
      showToast(`✓ Đã xác nhận "${name}"`);
      setTimeout(resetAll, 1200);
    }

    function handleReject() {
      showToast('Đã hủy bỏ');
      resetAll();
    }

    }
